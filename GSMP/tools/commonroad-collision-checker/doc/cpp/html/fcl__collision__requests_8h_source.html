279 bool defaultCollisionFunctionOverlap(fcl::CollisionObject<S> *o1,
280  fcl::CollisionObject<S> *o2,
281  void *cdata_) {
282  CollisionDataOverlap<S> *cdata =
283  static_cast<CollisionDataOverlap<S> *>(cdata_);
284  fcl::CollisionRequest<S> *request = &(cdata->request);
285  fcl::CollisionResult<S> *result = &(cdata->result);
286 
287  if (cdata->done)
288  return true;
289 
290  const auto &pair_obj =
291  std::pair<fcl::CollisionObject<S> *, fcl::CollisionObject<S> *>(o1, o2);
292 
293  // TODO: if collision detected earlier - don't check again!
294  auto &checked_pairs = cdata->self_reqData.checkedPairs;
295  auto search = checked_pairs.find(pair_obj);
296  if (search != checked_pairs.end())
297  return false;
298 
299  fcl::collide(o1, o2, *request, *result);
300 
301  checked_pairs.emplace(pair_obj);
302 
303  if (cdata->result.isCollision()) {
304  (cdata->result).clear();
305  const CollisionObject *obj1_entity = getParentPointerFromFclObj(o1);
306  const CollisionObject *obj2_entity = getParentPointerFromFclObj(o2);
307 
308  if (obj1_entity && obj2_entity) {
309  std::list<int> cont1_obj1_list;
310  std::list<int> cont1_obj2_list;
311  std::list<int> cont2_obj1_list;
312  std::list<int> cont2_obj2_list;
313 
314  int res_cont1_obj1 =
315  cdata->self_reqData.get_cont_1()->queryContainedObjectIndexList(
316  obj1_entity, cont1_obj1_list);
317  int res_cont1_obj2 =
318  cdata->self_reqData.get_cont_1()->queryContainedObjectIndexList(
319  obj2_entity, cont1_obj2_list);
320 
321  int res_cont2_obj1 =
322  cdata->self_reqData.get_cont_2()->queryContainedObjectIndexList(
323  obj1_entity, cont2_obj1_list);
324  int res_cont2_obj2 =
325  cdata->self_reqData.get_cont_2()->queryContainedObjectIndexList(
326  obj2_entity, cont2_obj2_list);
327 
328  if (!res_cont1_obj1 && !res_cont2_obj1) // obj1 is in both containers
329  {
330  cdata->self_reqData.addRequestResultList(cont1_obj1_list,
331  cont2_obj1_list);
332  }
333 
334  if (!res_cont1_obj2 && !res_cont2_obj2) // obj2 is in both containers
335  {
336  cdata->self_reqData.addRequestResultList(cont1_obj2_list,
337  cont2_obj2_list);
338  }
339  if (!res_cont1_obj1 &&
340  !res_cont2_obj2) // obj1 in container 1, obj2 in container 2
341  {
342  cdata->self_reqData.addRequestResultList(cont1_obj1_list,
343  cont2_obj2_list);
344  }
345 
346  if (!res_cont1_obj2 &&
347  !res_cont2_obj1) // obj1 in container 1, obj2 in container 2
348  {
349  cdata->self_reqData.addRequestResultList(cont1_obj2_list,
350  cont2_obj1_list);
351  }
352  }
353  }

358 template <typename S>
359 bool defaultCollisionFunctionListOfObstacles(fcl::CollisionObject<S> *o1,
360  fcl::CollisionObject<S> *o2,
361  void *cdata_) {
362  auto *cdata = static_cast<CollisionDataListOfObstacles<S> *>(cdata_);
363  const auto &request = cdata->request;
364 
365  if (cdata->done)
366  return true;
367 
368  fcl::CollisionObject<S> *obst = 0;
369 
370  fcl::CollisionObject<S> *subj = 0;
371 
372  // TODO: if collision detected earlier - don't check again!
373  if (cdata->self_reqData.getObstacleCallbackOrder()) {
374  obst = o1;
375  subj = o2;
376  } else {
377  subj = o1;
378  obst = o2;
379  }
380 
381  const auto &pair_obj =
382  std::pair<fcl::CollisionObject<S> *, fcl::CollisionObject<S> *>(obst,
383  subj);
384 
385  if (!(cdata->self_reqData.testPair(pair_obj)))
386  return false;
387 
388  collide(obst, subj, request, cdata->result);
389 
390  if (cdata->result.isCollision()) {
391  cdata->set_result(true);
392 
393  cdata->result.clear();
394 
395  const CollisionObject *obst_entity = getParentPointerFromFclObj(obst);
396  const CollisionObject *subj_entity = getParentPointerFromFclObj(subj);
397 
398  // try to add both, as the order in the callback is undefined
399  if (cdata->self_reqData.addRequestResultObstacle(obst_entity)) {
400  cdata->self_reqData.setAllObst_added();
401  return true;
402  }
403  if (cdata->self_reqData.addRequestResultObstacle(subj_entity)) {
404  cdata->self_reqData.setAllObst_added();
405  return true;
406  }
407  }
408 
409  return cdata->done;
410 }
