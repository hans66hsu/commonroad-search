110 void Triangle::compute_incircle_radius_and_center() {
111  // length of triangle sides
112  double a = (v2_ - v1_).norm();
113  double b = (v3_ - v2_).norm();
114  double c = (v3_ - v1_).norm();
115  // half of perimeter
116  double p = 0.5 * (a + b + c);
117  // using Heron's Formula
118  double area = std::sqrt(p * (p - a) * (p - b) * (p - c));
119  // incenter
120  incenter_ = (b * v1_ + c * v2_ + a * v3_) / (2.0 * p);
121  // inradius
122  incircle_radius_ = area / p;
