58 std::size_t collide_tvobst_obj(const CollisionObject &obj1,
59  const CollisionObject &obj2,
60  CollisionResult &res,
61  const CollisionRequest &req) {
62  static solvers::FCLSolver solver;
63  static CollisionFunctionMatrix matr_fcl(&solver);
64  const TimeVariantCollisionObject &tvobst(
65  static_cast<const TimeVariantCollisionObject &>(obj1));
66  int time_idx_start = tvobst.time_start_idx();
67  int time_idx_end = tvobst.time_end_idx();
68  for (int cc1 = time_idx_start; cc1 <= time_idx_end; cc1++) {
69  CollisionObjectConstPtr cur_obj = tvobst.getObstacleAtTime(cc1);
70 
71  if (!cur_obj.get()) {
72  continue;
73  }
74 
75  if (cur_obj->getCollisionObjectClass() ==
76  CollisionObjectClass::OBJ_CLASS_TVOBSTACLE) {
77  return -1;
78  }
79 
80  collide_bool_func_t func = matr_fcl.getSolverBoolFunction(
81  cur_obj->getCollisionObjectType(), obj2.getCollisionObjectType());
82  if (!func) {
83  return -1;
84  }
85  func(*cur_obj, obj2, res, req);
86  if (res.collides())
87  return 0;
88  }
89  return 0;
90 }

109  int time_idx_start =
110  std::min(tvobst1.time_start_idx(), tvobst2.time_start_idx());
111  int time_idx_end = std::max(tvobst1.time_end_idx(), tvobst2.time_end_idx());
112 
113  for (int cc1 = time_idx_start; cc1 <= time_idx_end; cc1++) {
114  CollisionObjectConstPtr obst_1 = tvobst1.getObstacleAtTime(cc1);
115  CollisionObjectConstPtr obst_2 = tvobst2.getObstacleAtTime(cc1);
116 
117  if (!obst_1.get() || !obst_2.get()) {
118  continue;
119  }
120  if (obst_1->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE ||
121  obst_2->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE) {
122  return -1;
123  }
124 
125  collide_bool_func_t func = matr_fcl.getSolverBoolFunction(
126  obst_1->getCollisionObjectType(), obst_2->getCollisionObjectType());
127  if (!func) {
128  return -1;
129  }
130  func(*obst_1, *obst_2, res, req);
131  if (res.collides())
132  return 0;
133  }
134  return 0;
