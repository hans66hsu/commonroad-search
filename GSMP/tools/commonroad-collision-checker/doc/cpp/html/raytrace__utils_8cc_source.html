204 int rayTraceRemoveOverlaps(std::vector<LineSegment> intersect,
205  std::vector<LineSegment> &out_vec, int axis) {
206  std::vector<LineSegment> starts = intersect;
207  std::vector<LineSegment> ends = intersect;
208  std::vector<bool> remove_its(intersect.size());
209  // todo zeromemory?
210 
211  for (auto &it : starts) {
212  if (it.point1().getCoord(axis) > it.point2().getCoord(axis)) {
213  it.swap();
214  }
215  }
216  if (!axis) {
217  std::sort(starts.begin(), starts.end(), startsSortX);
218  } else {
219  std::sort(starts.begin(), starts.end(), startsSortY);
220  }
221  // std::sort(ends.begin(), ends.end(), endsSortX);
222 
223  std::vector<LineSegment>::iterator starts_it = starts.begin();
224  while (starts_it < starts.end()) {
225  bool repeat = 1;
226  auto starts_it2 = starts_it + 1;
227 
228  while (repeat) {
229  std::vector<LineSegment>::iterator maxend = starts_it;
230  while ((starts_it2 < starts.end()) &&
231  (starts_it2)->point1().getCoord(axis) <=
232  (starts_it)->point2().getCoord(axis)) {
233  remove_its[starts_it2 - starts.begin()] = true;
234  if ((starts_it2)->point2().getCoord(axis) >=
235  maxend->point2().getCoord(axis)) {
236  maxend = starts_it2;
237  }
238  starts_it2++;
239  }
240  if (maxend != starts_it) {
241  starts_it->set_point_2(maxend->point2());
242  } else {
243  starts_it = starts_it2;
244  repeat = 0;
245  }
246  }
247  }
248 
249  for (unsigned int cc1 = 0; cc1 < starts.size(); cc1++) {
250  if (!remove_its[cc1]) {
251  out_vec.push_back(starts[cc1]);
252  }
253  }
254 
255  return 0;
