12 std::size_t collide_tvobst_obj(const CollisionObject &obj1,
13  const CollisionObject &obj2,
14  CollisionResult &res,
15  const CollisionRequest &req) {
16  static solvers::DefaultSolver solver;
17  static CollisionFunctionMatrix matr_default(&solver);
18  const TimeVariantCollisionObject &tvobst(
19  static_cast<const TimeVariantCollisionObject &>(obj1));
20  int time_idx_start = tvobst.time_start_idx();
21  int time_idx_end = tvobst.time_end_idx();
22  for (int cc1 = time_idx_start; cc1 <= time_idx_end; cc1++) {
23  CollisionObjectConstPtr cur_obj = tvobst.getObstacleAtTime(cc1);
24 
25  if (!cur_obj.get()) {
26  continue;
27  }
28 
29  if (cur_obj->getCollisionObjectClass() ==
30  CollisionObjectClass::OBJ_CLASS_TVOBSTACLE) {
31  return -1;
32  }
33 
34  collide_bool_func_t func = matr_default.getSolverBoolFunction(
35  cur_obj->getCollisionObjectType(), obj2.getCollisionObjectType());
36  if (!func) {
37  return -1;
38  }
39  func(*cur_obj, obj2, res, req);
40  if (res.collides())
41  return 0;
42  }
43  return 0;
44 }

60  static_cast<const TimeVariantCollisionObject &>(obj1));

62  static_cast<const TimeVariantCollisionObject &>(obj2));
63  int time_idx_start =
64  std::min(tvobst1.time_start_idx(), tvobst2.time_start_idx());
65  int time_idx_end = std::max(tvobst1.time_end_idx(), tvobst2.time_end_idx());
66 
67  for (int cc1 = time_idx_start; cc1 <= time_idx_end; cc1++) {
68  CollisionObjectConstPtr obst_1 = tvobst1.getObstacleAtTime(cc1);
69  CollisionObjectConstPtr obst_2 = tvobst2.getObstacleAtTime(cc1);
70 
71  if (!obst_1.get() || !obst_2.get()) {
72  continue;
73  }
74  if (obst_1->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE ||
75  obst_2->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE) {
76  return -1;
77  }
78 
79  collide_bool_func_t func = matr_default.getSolverBoolFunction(
80  obst_1->getCollisionObjectType(), obst_2->getCollisionObjectType());
81  if (!func) {
82  return -1;
83  }
84  func(*obst_1, *obst_2, res, req);
85  if (res.collides())
86  return 0;
87  }
88  return 0;
