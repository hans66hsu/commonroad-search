938

941  bool true_Failure = false;
942 
943  for (auto &obj : cc->getObstacles()) {
944

946  co->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE) {

953  }
954  if (obj->getCollisionObjectClass() == OBJ_CLASS_TVOBSTACLE) {
955  obj_tv = static_cast<const TimeVariantCollisionObject *>(obj.get());
956  }
957  if (obj_tv && co_tv) {
958  time_start_idx =
959  std::min(obj_tv->time_start_idx(), co_tv->time_start_idx());
960  time_end_idx = std::max(obj_tv->time_end_idx(), co_tv->time_end_idx());
961  } else {
962  if (obj_tv) {
963  time_start_idx = obj_tv->time_start_idx();
964  time_end_idx = obj_tv->time_end_idx();
965  } else if (co_tv) {
966  time_start_idx = co_tv->time_start_idx();
967  time_end_idx = co_tv->time_end_idx();
968  } else
969  throw;
970  }
971  for (int time_idx = time_start_idx; time_idx < time_end_idx; time_idx++) {
972 
973  CollisionObjectConstPtr obj1(nullptr);
974  if (obj_tv) {
975  obj1 = obj_tv->getObstacleAtTime(time_idx);
976  } else
977  obj1 = obj;
978  CollisionObjectConstPtr obj2(nullptr);
979  if (co_tv)
980  obj2 = co_tv->getObstacleAtTime(time_idx);
981  else
982  obj2 = co;
983  if (!obj1.get() || !obj2.get()) {
984  continue;
985  }
986 
987  bool tol_res = false;

989  ToleranceCheck(obj1, obj2, tol_res, dist);
990  if (!tol_res) {
991  tvgroup_tolerance_passed = false;
992  break;
993  }
994  }

996  result = true;
997  } else {
998  result = false;
999  }

1001  bool tol_res = false;
1002  double dist = 0;
1003  ToleranceCheck(obj, co, tol_res, dist);
1004  if (!tol_res) {
1005  result = false;
1006  }
1007  }

1010  break;
1011  }
1012  }
