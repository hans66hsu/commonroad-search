151 bool CollisionChecker::rayTrace(const Eigen::Vector2d &point1,
152  const Eigen::Vector2d &point2,
153  std::vector<LineSegment> &intersect,
154  bool remove_overlaps) {
155  double eps = 0.000001;
156  double sqrlen =
157  (pow(point1[0] - point2[0], 2) + pow(point1[1] - point2[1], 2));
158  // if ray length is greater than epsilon, filter the colliding obstacles
159  if (sqrlen >= eps) {
160  Eigen::Vector2d center = (point2 + point1) / 2;
161  Eigen::Vector2d local_x_axis = center - point1;
162 
163  double rad_x = sqrt(sqrlen) / 2; // radius of the ray OBB
164  local_x_axis /= rad_x;
165  Eigen::Vector2d local_y_axis(-local_x_axis[1], local_x_axis[0]);
166  Eigen::Matrix2d local_axes;
167  local_axes.col(0) = local_x_axis;
168  local_axes.col(1) = local_y_axis;
169  collision::RectangleOBBConstPtr robb(
170  new collision::RectangleOBB(rad_x, eps, local_axes, center));
171  std::vector<CollisionObjectConstPtr> obstacles;
172 
173  if (collide(robb, obstacles, 1, 1)) {
174 
175  return collision::raytrace::rayTracePrimitive(obstacles, point1, point2,
176  intersect, remove_overlaps);
177  } else
178  return false;
179 
180  } else {
181  return collision::raytrace::rayTracePrimitive(
182  collision_objects_, point1, point2, intersect, remove_overlaps);
183  }
