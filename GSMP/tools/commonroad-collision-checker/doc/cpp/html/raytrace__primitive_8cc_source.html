24 // http://csharphelper.com/blog/2014/09/determine-where-a-line-intersects-a-circle-in-c/
25 int findLineCircleIntersections(double cx, double cy, double radius,
26  const Eigen::Vector2d &p1,
27  const Eigen::Vector2d &p2,
28  std::vector<Eigen::Vector2d> &inters) {
29  double dx, dy, A, B, C, det, t;
30 
31  dx = p2.x() - p1.x();
32  dy = p2.y() - p1.y();
33 
34  A = dx * dx + dy * dy;
35  B = 2 * (dx * (p1.x() - cx) + dy * (p1.y() - cy));
36  C = (p1.x() - cx) * (p1.x() - cx) + (p1.y() - cy) * (p1.y() - cy) -
37  radius * radius;
38 
39  det = B * B - 4 * A * C;
40  if ((A <= 0.0000001) || (det < 0)) {
41  // No real solutions.
42  return 0;
43  } else if (det == 0) {
44  // One solution.
45  t = -B / (2 * A);
46  inters.push_back(Eigen::Vector2d(p1.x() + t * dx, p1.y() + t * dy));
47  return 1;
48  } else {
49  // Two solutions.
50  t = (double)((-B + sqrt(det)) / (2 * A));
51  inters.push_back(Eigen::Vector2d(p1.x() + t * dx, p1.y() + t * dy));
52 
53  t = (double)((-B - sqrt(det)) / (2 * A));
54  inters.push_back(Eigen::Vector2d(p1.x() + t * dx, p1.y() + t * dy));
55  return 2;
56  }
57 }

81 // To find orientation of ordered triplet (p, q, r).
82 // The function returns following values
83 // 0 --> p, q and r are colinear
84 // 1 --> Clockwise
85 // 2 --> Counterclockwise
86 int orientation(Point p, Point q, Point r) {
87  // See 10th slides from following link for derivation of the formula
88  // http://www.dcs.gla.ac.uk/~pat/52233/slides/Geometry1x1.pdf
89  double val = (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y);
90 
91  if (val == 0)
92  return 0; // colinear
93 
94  return (val > 0) ? 1 : 2; // clock or counterclock wise
95 }

101 bool lineLineIntersect(double x1, double y1, // Line 1 start
102  double x2, double y2, // Line 1 end
103  double x3, double y3, // Line 2 start
104  double x4, double y4, // Line 2 end
105  double &ixOut, double &iyOut) // Output
106 {
107  // http://mathworld.wolfram.com/Line-LineIntersection.html
108 
109  double detL1 = det(x1, y1, x2, y2);
110  double detL2 = det(x3, y3, x4, y4);
111  double x1mx2 = x1 - x2;
112  double x3mx4 = x3 - x4;
113  double y1my2 = y1 - y2;
114  double y3my4 = y3 - y4;
115 
116  double xnom = det(detL1, x1mx2, detL2, x3mx4);
117  double ynom = det(detL1, y1my2, detL2, y3my4);
118  double denom = det(x1mx2, y1my2, x3mx4, y3my4);
119  if (denom == 0.0) // Lines don't seem to cross
120  {
121  ixOut = NAN;
122  iyOut = NAN;
123  return false;
124  }
125 
126  ixOut = xnom / denom;
127  iyOut = ynom / denom;
128  if (!isfinite(ixOut) || !isfinite(iyOut)) // Probably a numerical issue
129  return false;
130 
131  return true; // All OK
132 }

143  int o3 = orientation(p2, q2, p1);

145 
146  // General case
147  if (o1 != o2 && o3 != o4) {
148  double int_x = 0;
149  double int_y = 0;
150  if (lineLineIntersect(p1.x, p1.y, q1.x, q1.y, p2.x, p2.y, q2.x, q2.y, int_x,
151  int_y)) {

154  intersection1.y = int_y;
155  inters.push_back(intersection1);
156  return true;

159  return false;
160  }
161  }
162

164  if (o1 == 0 && o2 == 0) {
165  int outnum = 0;
166 
167  // if p2 or q2 lies on p1q1
168

179  }
180

182  return true;
183 
184  // if p1 or q1 lies on p2q2
185

187  outnum++;
188  inters.push_back(p1);
189  }
190

192  return true;
193

195  outnum++;
196  inters.push_back(q1);

200  return true;
201  }
202

206  inters.push_back(p2);
